// The robot will try to match the sentence with one of the templates below.
// If there's a match, the specified function will deal with the rest of the processing.
// The first 2 numbers indicate which words are passed to the function.
//
// Note, where there ia a *, no match is needed

// CLASS STATEMENTS
// parameters: subject, class
a * is a *,  is_statement(2, 5)               // ex: a cat is an animal
* is a *,    is_statement(1, 4)               // ex: beer is a beverage

// CLASS QUESTIONS
// parameters: subject, class
what is *,   is_question(3)          // ex: what is coffee?
what are *,  is_question(3)          // ex: what are raccoons?
what is a *, is_question(4)          // ex: what is a cow?

// ATTRIBUTE STATEMENTS
// parameters: subject, attribute
* is *,      is_statement(1, 3)               // ex: chocolate is brown

// ATTRIBUTE QUESTIONS
// parameters: subject, attribute
what * is *, is_question(4, 2)                // ex: what color is chocolate
where is *,  is_question(3, "location")       // ex: where is lucy?

// ATTRIBUTE CONFIRMAION QUESTIONS
// parameters: subject, attribute
is * *, is_confirmation_question(2, 3)        // ex: is chocolate brown
